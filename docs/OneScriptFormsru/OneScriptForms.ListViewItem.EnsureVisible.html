ЭлементСпискаЭлементов.ОбеспечитьОтображение (ListViewItem.EnsureVisible) Метод

Синтаксис

ОбеспечитьОтображение()

Параметры

Возвращаемое значение

Описание

Гарантирует видимость элемента списка элементов выполнением прокрутки содержимого элемента управления.

Примечание

Этот метод можно использовать для гарантированного отображения элемента в элементе управления СписокЭлементов (ListView).

Пример

СписокЭлементов1.ОбеспечитьОтображение(45);
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Ширина = 200;
Форма1.Высота = 150;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

СписокЭлементов1 = Форма1.ЭлементыУправления.Добавить(Ф.СписокЭлементов());
СписокЭлементов1.Стыковка = Ф.СтильСтыковки.Заполнение;
СписокЭлементов1.Флажки = Истина;
СписокЭлементов1.РежимОтображения = Ф.РежимОтображения.Подробно;
СписокЭлементов1.МножественныйВыбор = Истина;

Колонка1 = СписокЭлементов1.Колонки.Добавить(Ф.Колонка("Номер", 70, Ф.ГоризонтальноеВыравнивание.Лево));
Элементы = СписокЭлементов1.Элементы;
Для А = 1 По 100 Цикл
	ТекстЭлемента = "00" + А;
	ТекстЭлемента = Прав(ТекстЭлемента, 3);
	Элементы.Добавить(Ф.ЭлементСпискаЭлементов(ТекстЭлемента));
КонецЦикла;
СписокЭлементов1.Сортировать(Колонка1, Ф.ПорядокСортировки.ПоВозрастанию);

СписокЭлементов1.Фокус();

СписокЭлементов1.Элементы(45).Помечен = Истина;

СписокЭлементов1.ОбеспечитьОтображение(45);
	
Ф.ЗапуститьОбработкуСобытий();
