СтильТаблицыСеткиДанных.СтилиКолонкиСеткиДанных (DataGridTableStyle.GridColumnStyles) Свойство

Использование

Только чтение.

Значение

Тип: СтилиКолонкиСеткиДанных (GridColumnStylesCollection).

Возвращает коллекцию колонок, нарисованных для этой таблицы.

Примечание

Пример

СтилиКолонкиСеткиДанных1 = СтильТаблицыСеткиДанных1.СтилиКолонкиСеткиДанных;
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Ширина = 400;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

СеткаДанных1 = Ф.СеткаДанных();
СеткаДанных1.Родитель = Форма1;
СеткаДанных1.ТекстЗаголовка = "Сетка данных";
СеткаДанных1.Стыковка = Ф.СтильСтыковки.Заполнение;
	
ТаблицаДанных1 = Ф.ТаблицаДанных("ТД1");
Колонка1 = ТаблицаДанных1.Колонки.Добавить(Ф.КолонкаДанных("ФИО", Ф.ТипДанных.Строка));

Стр13 = "Трифонов Юрий Никифорович

	ТаблицаДанных1.Строки.Добавить(ТаблицаДанных1.НоваяСтрока());
КонецЦикла;

Для А = 1 По СтрЧислоСтрок(Стр13) Цикл
	Стр14 = СтрПолучитьСтроку(Стр13, А);
	М9 = СтрРазделить(Стр14, ",");
	М_ФИО.Добавить(М9[0]);
КонецЦикла;
ТаблицаДанных1.ЗагрузитьКолонку(М_ФИО, 0);

СтильТаблицыСеткиДанных1 = Ф.СтильТаблицыСеткиДанных();
СтильТаблицыСеткиДанных1.ИмяОтображаемого = "ТД1";

СтилиТаблицы1 = СеткаДанных1.СтилиТаблицы;
ИндексСтильТаблицыСеткиДанных1 = СтилиТаблицы1.Добавить(СтильТаблицыСеткиДанных1);

СтильКолонки1 = Ф.СтильКолонкиПолеВвода();
СтильКолонки1.ИмяОтображаемого = "ФИО";
СтильКолонки1.ТекстЗаголовка = "ФИО";
СтильКолонки1.Ширина = 250;
СтильТаблицыСеткиДанных1.СтилиКолонкиСеткиДанных.Добавить(СтильКолонки1);

СеткаДанных1.ИсточникДанных = ТаблицаДанных1;

Ф.ЗапуститьОбработкуСобытий();
