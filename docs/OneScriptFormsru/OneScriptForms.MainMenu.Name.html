ГлавноеМеню.Имя (MainMenu.Name) Свойство

Использование

Чтение и запись.

Значение

Тип: Строка.

Возвращает или задает имя элемента управления.

Примечание

Пример

ГлавноеМеню1.Имя = "ГлавноеМеню1";
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Текст = "Пример Главного меню";
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

ГлавноеМеню1 = Ф.ГлавноеМеню();
ГлавноеМеню1.Имя = "ГлавноеМеню1";
_Файл = ГлавноеМеню1.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Файл"));
_Справка = ГлавноеМеню1.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Справка"));

_Новый = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Новый"));
_Открыть = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Открыть"));
_СохранитьКак = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Сохранить как..."));

_txt = _СохранитьКак.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Файл *.txt"));
_rtf = _СохранитьКак.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Файл *.rtf"));

_Помошник = _Справка.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Помошник"));
_Опрограмме = _Справка.ЭлементыМеню.Добавить(Ф.ЭлементМеню("О программе..."));

Форма1.Меню = ГлавноеМеню1;

Сообщить("ГлавноеМеню1.Имя = " + ГлавноеМеню1.Имя);

Ф.ЗапуститьОбработкуСобытий();
