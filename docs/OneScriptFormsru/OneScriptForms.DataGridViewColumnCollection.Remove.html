КолонкиТаблицы.Удалить (DataGridViewColumnCollection.Remove) Метод

Синтаксис

Удалить(<Колонка>)

Параметры

Колонка (обязательный)
Тип: КолонкаТаблицы (DataGridViewColumn).
Удаляемая колонка.

Возвращаемое значение

Описание

Удаляет указанную колонку из коллекции.

Примечание

Пример

Таблица1.Колонки.Удалить(КолонкаПолеВвода);
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();

Форма1 = Ф.Форма();
Форма1.Ширина = 500;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

Таблица1 = Ф.Таблица();
Таблица1.Родитель = Форма1;
Таблица1.Стыковка = Ф.СтильСтыковки.Заполнение;
Таблица1.АвтонумерацияСтрок = Истина;

Для А = 0 По 20 Цикл
	Таблица1.Колонки.Добавить(Ф.КолонкаПолеВвода());
	Таблица1.Колонки(А).ТекстЗаголовка = "Кол" + А;
КонецЦикла;
Для А = 0 По 20 Цикл
	Таблица1.Строки.Добавить();
КонецЦикла;

КолонкаПолеВвода = Ф.КолонкаПолеВвода();
КолонкаПолеВвода.ТекстЗаголовка = "Вставленная";
Таблица1.Колонки.Вставить(1, КолонкаПолеВвода);
Таблица1.Колонки.Удалить(КолонкаПолеВвода);

Ф.ЗапуститьОбработкуСобытий();
