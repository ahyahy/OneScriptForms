ЭлементМеню.Помечен (MenuItem.Checked) Свойство

Использование

Чтение и запись.

Значение

Тип: Булево.

Возвращает или задает значение, указывающее, что установлен флажок рядом с текстом пункта меню.

Примечание

Это свойство можно использовать для задания выбранного пункта меню в группе взаимоисключающих пунктов меню.

Значение Истина допустимо только для элемента меню, не имеющего дочерних элементов и не принадлежащего к верхнему уровню.

Пример

Подменю_Файл_Открыть.Помечен = Истина;
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

ГлавноеМеню1 = Ф.ГлавноеМеню();
_Файл = ГлавноеМеню1.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Файл"));
_Справка = ГлавноеМеню1.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Справка"));

Подменю_Файл_Новый = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Новый"));
Подменю_Файл_Открыть = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Открыть"));
Подменю_Файл_Сохранить = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Сохранить"));
Подменю_Файл_СохранитьКак = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Сохранить как..."));
Подменю_Файл_Выход = _Файл.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Выход"));

Подменю_Файл_СохранитьКак_txt = Подменю_Файл_СохранитьКак.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Файл *.txt"));
Подменю_Файл_СохранитьКак_rtf = Подменю_Файл_СохранитьКак.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Файл *.rtf"));

Подменю_Справка_Помошник = _Справка.ЭлементыМеню.Добавить(Ф.ЭлементМеню("Помошник"));
Подменю_Справка_Опрограмме = _Справка.ЭлементыМеню.Добавить(Ф.ЭлементМеню("О программе..."));

Форма1.Меню = ГлавноеМеню1;

Подменю_Файл_Открыть.Переключатель = Истина;
Подменю_Файл_Открыть.Помечен = Истина;

Ф.ЗапуститьОбработкуСобытий();
