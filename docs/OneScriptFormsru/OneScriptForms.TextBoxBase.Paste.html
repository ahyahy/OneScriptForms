ПолеВводаОснова.Вставить (TextBoxBase.Paste) Метод

Синтаксис

Вставить()

Параметры

Возвращаемое значение

Описание

Заменяет текущее выделение в ПолеВвода (TextBox) содержимым буфера обмена.

Примечание

Пример

ПолеВвода1.Вставить();
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Ширина = 750;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

ПолеВвода1 = Форма1.ЭлементыУправления.Добавить(Ф.ПолеВвода());
ПолеВвода1.Текст = "Возвращает или задает значение, указывающее текст, выбранный в настоящий момент в элементе управления.";
ПолеВвода1.Стыковка = Ф.СтильСтыковки.Заполнение;
ПолеВвода1.МногострочныйРежим = Истина;

ПолеВвода1.НачалоВыделения = 11;
ПолеВвода1.ДлинаВыделения = 3;

Буфер1 = Ф.БуферОбмена();
Если Ф.ПлатформаWin Тогда
	Буфер1.УстановитьТекст("ддд");
Иначе
	Буфер1.УстановитьТекст("ddd");
КонецЕсли;

ПолеВвода1.НачалоВыделения = 11;
ПолеВвода1.ДлинаВыделения = 3;

ПолеВвода1.Вставить();
ПолеВвода1.НачалоВыделения = 11;
ПолеВвода1.ДлинаВыделения = 3;

ПолеВвода1.Фокус();

Ф.ЗапуститьОбработкуСобытий();
