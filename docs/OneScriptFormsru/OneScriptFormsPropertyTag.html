Свойство Метка у контрола

Свойство Метка расширяет возможности объекта с помощью дополнительных данных с различными типами значений.

Привожу пример использования метки.

Пример

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

Реквизиты = Новый Структура("Поле1","ЧтоТоТам");

ПолеВвода1 = Форма1.ЭлементыУправления.Добавить(Ф.ПолеВвода());
ПолеВвода1.Имя = "Поле1";

Сообщить("ТипЗнч(ПолеВвода1.Метка) = " + ТипЗнч(ПолеВвода1.Метка));
ПолеВвода1.Метка.Добавить(Реквизиты, "Реквизиты");// установили значение в коллекции по индексу 0

НашаСтруктура = ПолеВвода1.Метка.Элемент(0);
Знач1 = НашаСтруктура["Поле1"];
Сообщить("Знач1 = " + Знач1);

НашаСтруктура["Поле1"] = "ДругоеЧтоТоТам";
Знач2 = НашаСтруктура["Поле1"];
Сообщить("Знач2 = " + Знач2);

Пока Ф.Продолжать Цикл
	Выполнить(Ф.ПолучитьСобытие());
КонецЦикла;
