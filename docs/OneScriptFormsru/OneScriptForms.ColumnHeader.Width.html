Колонка.Ширина (ColumnHeader.Width) Свойство

Использование

Чтение и запись.

Значение

Тип: Число.

Возвращает или задает ширину колонки.

Примечание

Пример

Колонка2.Ширина = 100;
Полный пример кода 




Перем Ф, СписокЭлементов1, Колонки1;

Функция Заполнить()
	СписокЭлементов1.ПриостановитьРазмещение();
	
	Колонки1 = СписокЭлементов1.Колонки;
	Элементы1 = СписокЭлементов1.Элементы;
	Колонки1.Очистить();
	Элементы1.Очистить();

	Колонки1.Добавить(Ф.Колонка("Объект", 250));
	
	Колонка2 = Ф.Колонка("Вес (кг)", , 1);
	Колонка2.Ширина = 100;
	Колонки1.Добавить(Колонка2);
	
	Элемент1 = Ф.ЭлементСпискаЭлементов("Велосипед Урал", 1);
	Элемент1.ИспользоватьСтильДляПодэлементов = Ложь;
	Элементы1.Добавить(Элемент1);
	Подэлементы1 = Элемент1.Подэлементы;
	Подэлемент1 = Ф.ПодэлементСпискаЭлементов("17.5");
	Подэлементы1.Добавить(Подэлемент1);
	
	Элемент2 = Ф.ЭлементСпискаЭлементов("Пароход Титаник", 1);
	Элемент2.ИспользоватьСтильДляПодэлементов = Ложь;
	Элементы1.Добавить(Элемент2);
	Подэлементы2 = Элемент2.Подэлементы;
	Подэлемент2 = Ф.ПодэлементСпискаЭлементов("46328000");
	Подэлементы2.Добавить(Подэлемент2);
	
	Элемент3 = Ф.ЭлементСпискаЭлементов("Самолет Боинг", 1);
	Элемент3.ИспользоватьСтильДляПодэлементов = Ложь;
	Элементы1.Добавить(Элемент3);
	Подэлементы3 = Элемент3.Подэлементы;
	Подэлемент3 = Ф.ПодэлементСпискаЭлементов("162000");
	Подэлементы3.Добавить(Подэлемент3);
	
	Элемент4 = Ф.ЭлементСпискаЭлементов("Муравей", 1);
	Элемент4.ИспользоватьСтильДляПодэлементов = Ложь;
	Элементы1.Добавить(Элемент4);
	Подэлементы4 = Элемент4.Подэлементы;
	Подэлемент4 = Ф.ПодэлементСпискаЭлементов("0.00015");
	Подэлементы4.Добавить(Подэлемент4);
	
	СписокЭлементов1.ВозобновитьРазмещение();
КонецФункции
