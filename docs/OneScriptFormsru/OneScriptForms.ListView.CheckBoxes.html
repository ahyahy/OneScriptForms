СписокЭлементов.Флажки (ListView.CheckBoxes) Свойство

Использование

Чтение и запись.

Значение

Тип: Булево.

Возвращает или задает значение, указывающее, будет ли отображаться флажок рядом с каждым элементом.

Примечание

Пример

СписокЭлементов1.Флажки = Истина;
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

СписокЭлементов1 = Форма1.ЭлементыУправления.Добавить(Ф.СписокЭлементов());
СписокЭлементов1.Стыковка = Ф.СтильСтыковки.Заполнение;
СписокЭлементов1.РежимОтображения = Ф.РежимОтображения.Подробно;

Колонка1 = Ф.Колонка("Имя цвета", 250, Ф.ГоризонтальноеВыравнивание.Центр);
Колонка1.ВыравниваниеТекста = Ф.ГоризонтальноеВыравнивание.Центр;
Колонка1.Текст = "Имя цвета";
Колонка1.Ширина = Колонка1.Ширина + 10;
СписокЭлементов1.Колонки.Добавить(Колонка1);

М3 = СтрРазделить(Ф.СвойстваОбъекта(Ф.Цвет()), Символы.ПС);
Для А = 0 По 20 Цикл
	М4 = СтрРазделить(М3[А], " ");
	ИмяЦвета = М4[0];
	Попытка
		Цв = Ф.Цвет(ИмяЦвета);
	Исключение
		Продолжить;
	КонецПопытки;
	СписокЭлементов1.Элементы.Добавить(Ф.ЭлементСпискаЭлементов(ИмяЦвета));
КонецЦикла;

СписокЭлементов1.Флажки = Истина;

Ф.ЗапуститьОбработкуСобытий();
