Заполнение.Имя (Padding.Name) Свойство

Использование

Чтение и запись.

Значение

Тип: Строка.

Возвращает или задает имя элемента управления.

Примечание

Пример

Заполнение1.Имя = "Заполнение1";
Сообщить("Заполнение1.Имя = " + Заполнение1.Имя);
Полный пример кода 



ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();

Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

Таблица1 = Ф.Таблица();
Таблица1.Стыковка = Ф.СтильСтыковки.Заполнение;
Таблица1.Родитель = Форма1;
Таблица1.КоличествоКолонок = 2;
Таблица1.КоличествоСтрок = 5;

Заполнение1 = Ф.Заполнение();
Заполнение2 = Ф.Заполнение(4);
Заполнение3 = Ф.Заполнение(4, 3, 4, 3);

СтильЯчейки1 = Ф.СтильЯчейки();
СтильЯчейки1.Заполнение = Заполнение2;
СтильЯчейки1.Выравнивание = Ф.ВыравниваниеСодержимогоЯчейки.СерединаЛево;
СтильЯчейки1.ЦветФона = Ф.Цвет("Красный");
Таблица1.СтильЗаголовковСтрок = СтильЯчейки1;

Таблица1.Строки(0).Ячейки(0).Значение = "Ячейка1";
Таблица1.Строки(1).ЗаголовокСтроки.Значение = "Вторая";

Таблица1.Колонки(0).ЗаголовокКолонки.Значение = "Колонка1";
Таблица1.Колонки(1).ЗаголовокКолонки.Значение = "Колонка2";

Заполнение1.Имя = "Заполнение1";
Сообщить("Заполнение1.Имя = " + Заполнение1.Имя);
