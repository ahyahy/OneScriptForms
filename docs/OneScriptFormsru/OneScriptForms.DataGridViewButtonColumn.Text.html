КолонкаКнопка.Текст (DataGridViewButtonColumn.Text) Свойство

Использование

Чтение и запись.

Значение

Тип: Строка.

Возвращает или задает текст, отображаемый на ячейке кнопки по умолчанию.

Примечание

Пример

КолонкаКнопка1.Текст = "Кнопка";
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();

Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

Таблица1 = Ф.Таблица();
Таблица1.Стыковка = Ф.СтильСтыковки.Заполнение;
Таблица1.Родитель = Форма1;
Таблица1.АвтоНумерацияСтрок = Истина;

КолонкаКнопка1 = Ф.КолонкаКнопка();
Таблица1.Колонки.Добавить(КолонкаКнопка1);

Таблица1.ДобавлятьСтроки = Ложь;
Таблица1.КоличествоСтрок = 5;

КолонкаКнопка1.Ширина = 140;
