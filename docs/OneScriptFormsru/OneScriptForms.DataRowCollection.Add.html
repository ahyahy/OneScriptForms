СтрокиДанных.Добавить (DataRowCollection.Add) Метод

Синтаксис

Добавить(<СтрокаДанных>)

Параметры

СтрокаДанных (обязательный)
Тип: СтрокаДанных (DataRow).

Возвращаемое значение

Тип: СтрокаДанных (DataRow).

Описание

Создает и добавляет указанный объект СтрокаДанных (DataRow) в коллекцию.

Примечание

Пример

НоваяСтрока = СтрокиДанных1.Добавить(ТаблицаДанных1.НоваяСтрока());

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Ширина = 400;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

СеткаДанных1 = Ф.СеткаДанных();
СеткаДанных1.Родитель = Форма1;
СеткаДанных1.ТекстЗаголовка = "Сетка данных";
СеткаДанных1.Стыковка = Ф.СтильСтыковки.Заполнение;
	
ТаблицаДанных1 = Ф.ТаблицаДанных("ТД1");
Колонка1 = ТаблицаДанных1.Колонки.Добавить(Ф.КолонкаДанных("ФИО", Ф.ТипДанных.Строка));

СтрокиДанных1 = ТаблицаДанных1.Строки;
НоваяСтрока = СтрокиДанных1.Добавить(ТаблицаДанных1.НоваяСтрока());
НоваяСтрока.Элемент(0).Значение = "Алиса Игоревна Селезнева";
НоваяСтрока = СтрокиДанных1.Добавить(ТаблицаДанных1.НоваяСтрока());
НоваяСтрока.Элемент(0).Значение = "Эраст Петрович Фандорин";
НоваяСтрока = СтрокиДанных1.Добавить(ТаблицаДанных1.НоваяСтрока());
НоваяСтрока.Элемент(0).Значение = "Врунгель Христофор Бонифатьевич";

СтильТаблицыСеткиДанных1 = Ф.СтильТаблицыСеткиДанных();
СтильТаблицыСеткиДанных1.ИмяОтображаемого = "ТД1";
СтильТаблицыСеткиДанных1.ПредпочтительнаяШиринаСтолбцов = 280;
СтилиТаблицы1 = СеткаДанных1.СтилиТаблицы;
ИндексСтильТаблицыСеткиДанных1 = СтилиТаблицы1.Добавить(СтильТаблицыСеткиДанных1);

СеткаДанных1.ИсточникДанных = ТаблицаДанных1;
