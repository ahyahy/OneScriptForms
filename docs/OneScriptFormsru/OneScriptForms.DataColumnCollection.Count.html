КолонкиДанных.Количество (DataColumnCollection.Count) Свойство

Использование

Только чтение.

Значение

Тип: Число.

Возвращает общее количество колонок в коллекции.

Примечание

Пример

Сообщить("Колонки1.Количество = " + Колонки1.Количество);
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

ТаблицаДанных1 = Ф.ТаблицаДанных();
ТаблицаДанных1.ИмяТаблицы = "ТД1";
Колонки1 = ТаблицаДанных1.Колонки;
Сообщить("Колонки1.Количество = " + Колонки1.Количество); // Колонки1.Количество = 0
КолонкаДанных1 = Ф.КолонкаДанных();
Колонка1 = Колонки1.Добавить(КолонкаДанных1);
Колонка1.ТипДанных = Ф.ТипДанных.Число;
Колонка1.ИмяКолонки = "№ н/п";
Сообщить("Колонки1.Количество = " + Колонки1.Количество); // Колонки1.Количество = 1
