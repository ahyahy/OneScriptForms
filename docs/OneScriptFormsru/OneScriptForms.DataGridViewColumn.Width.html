КолонкаТаблицы.Ширина (DataGridViewColumn.Width) Свойство

Использование

Чтение и запись.

Значение

Тип: Число.

Возвращает или задает текущую ширину колонки.

Примечание

Пример

КолонкаПолеВвода1.Ширина = 200;
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();

Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

Таблица1 = Ф.Таблица();
Таблица1.Родитель = Форма1;
Таблица1.Стыковка = Ф.СтильСтыковки.Заполнение;
Таблица1.АвтоНумерацияСтрок = Истина;

КолонкаПолеВвода1 = Ф.КолонкаПолеВвода();
КолонкаПолеВвода1.ТекстЗаголовка = "Города";

КолонкаПолеВвода1.Ширина = 200;

Таблица1.Колонки.Добавить(КолонкаПолеВвода1);

Таблица1.КоличествоСтрок = 5;
Таблица1.Ячейка(0, 0).Значение = "Москва";
Таблица1.Ячейка(0, 1).Значение = "Париж";
Таблица1.Ячейка(0, 2).Значение = "Саратов";
Таблица1.Ячейка(0, 3).Значение = "Тюмень";
Таблица1.Ячейка(0, 4).Значение = "Хабаровск";

Ф.ЗапуститьОбработкуСобытий();
