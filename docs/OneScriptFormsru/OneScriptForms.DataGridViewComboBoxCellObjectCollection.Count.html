ЭлементыПоляВыбораЯчейки.Количество (DataGridViewComboBoxCellObjectCollection.Count) Свойство

Использование

Только чтение.

Значение

Тип: Число.

Возвращает количество элементов в коллекции.

Примечание

Пример

Сообщить("" + ЭлементыПоляВыбораЯчейки.Количество);
Полный пример кода 




ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();

Форма1 = Ф.Форма();
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

Таблица1 = Ф.Таблица();
Таблица1.Родитель = Форма1;
Таблица1.Стыковка = Ф.СтильСтыковки.Заполнение;
Таблица1.АвтоНумерацияСтрок = Истина;

КолонкаПолеВыбора1 = Ф.КолонкаПолеВыбора();
КолонкаПолеВыбора1.Ширина = 215;
Таблица1.Колонки.Добавить(КолонкаПолеВыбора1);
Таблица1.Колонки.Добавить(Ф.КолонкаПолеВыбора());
Таблица1.КоличествоСтрок = 3;

ЭлементыПоляВыбораЯчейки = КолонкаПолеВыбора1.Элементы;
ЭлементыПоляВыбораЯчейки.Добавить("Челябинск");
ЭлементыПоляВыбораЯчейки.Добавить("Хабаровск");
ЭлементыПоляВыбораЯчейки.Добавить("Тюмень");
ЭлементыПоляВыбораЯчейки.Добавить("Москва");
ЭлементыПоляВыбораЯчейки.Добавить("Саратов"); 

Таблица1.Строки(0).Ячейки(0).Значение = "Тюмень";
