КолонкаТаблицы.ВесЗаполнения (DataGridViewColumn.FillWeight) Свойство

Использование

Чтение и запись.

Значение

Тип: Число.

Возвращает или задает значение, представляющее ширину колонки, для которой свойство РежимАвтоРазмера (AutoSizeMode) установлено в значение Заполнение (Fill), относительно ширины других колонок элемента управления, находящихся в этом режиме.

Значение по умолчанию - 100.

Примечание

Пример

Колонка.ВесЗаполнения = 20;
Полный пример кода 




Перем Таблица1;

Функция Форма1_РазмерИзменен() Экспорт
	Таблица1.Обновить();
КонецФункции

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();

Форма1 = Ф.Форма();
Форма1.Ширина = 500;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();
Форма1.РазмерИзменен = Ф.Действие(ЭтотОбъект, "Форма1_РазмерИзменен");

Таблица1 = Ф.Таблица();
Таблица1.Родитель = Форма1;
Таблица1.Стыковка = Ф.СтильСтыковки.Заполнение;
Таблица1.АвтоНумерацияСтрок = Истина;
Таблица1.ДобавлятьСтроки = Ложь;
Таблица1.АвтоШиринаЗаголовковСтрок = Ф.РежимШириныЗаголовковСтрок.ДляОтображаемых;

Для А = 0 По 3 Цикл
	КолонкаПолеВвода = Ф.КолонкаПолеВвода();
	КолонкаПолеВвода.РежимАвтоРазмера = Ф.РежимАвтоРазмераКолонки.Заполнение;
	КолонкаПолеВвода.ТекстЗаголовка = "Кол" + А;
	Таблица1.Колонки.Добавить(КолонкаПолеВвода);
КонецЦикла;
Таблица1.КоличествоСтрок = 100000;

Колонка = Таблица1.Колонки(1);
Колонка.ВесЗаполнения = 20;

Ф.ЗапуститьОбработкуСобытий();
