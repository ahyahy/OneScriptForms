Синтаксис

Таймер()

Параметры

Возвращаемое значение

Тип: Таймер (Timer).

Описание

Создаёт новый экземпляр класса Таймер (Timer).

Примечание

Пример

Таймер1 = Ф.Таймер();

Перем Ф, КоличЦикловТаймера, Надпись2, Кнопка1, Таймер1, РегуляторВверхВниз1;

Процедура Кнопка1_Нажатие() Экспорт
    Если Кнопка1.Текст = "Таймер пуск" Тогда
		Таймер1.Интервал = РегуляторВверхВниз1.Значение * 1000;
		КоличЦикловТаймера = 0;
	    Кнопка1.Текст = "Таймер стоп";
		Надпись2.Текст = "" + КоличЦикловТаймера;
		РегуляторВверхВниз1.Доступность = Ложь;
		Таймер1.Начать();
	ИначеЕсли Кнопка1.Текст = "Таймер стоп" Тогда
		Таймер1.Остановить();
		КоличЦикловТаймера = 0;
	    Кнопка1.Текст = "Таймер пуск";
		РегуляторВверхВниз1.Доступность = Истина;
	КонецЕсли;
КонецПроцедуры

Процедура Таймер1_ПриСрабатыванииТаймера() Экспорт
	КоличЦикловТаймера = КоличЦикловТаймера + 1;
	Надпись2.Текст = "" + КоличЦикловТаймера;

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptForms\OneScriptForms\bin\Debug\OneScriptForms.dll");
Ф = Новый ФормыДляОдноСкрипта();
Форма1 = Ф.Форма();
Форма1.Ширина = 450;
Форма1.Высота = 200;
Форма1.Отображать = Истина;
Форма1.Показать();
Форма1.Активизировать();

КоличЦикловТаймера = 0;
Кнопка1 = Форма1.ЭлементыУправления.Добавить(Ф.Кнопка());
Кнопка1.Границы = Ф.Прямоугольник(10, 10, 150, 50);
Кнопка1.Текст = "Таймер пуск";
Кнопка1.Нажатие = Ф.Действие(ЭтотОбъект, "Кнопка1_Нажатие");

Надпись1 = Форма1.ЭлементыУправления.Добавить(Ф.Надпись());
Надпись1.Границы = Ф.Прямоугольник(Кнопка1.Лево, Кнопка1.Низ + 5, Кнопка1.Ширина, 40);
Надпись1.Текст = "секунд до срабатывания";
Надпись1.ВыравниваниеТекста = Ф.ВыравниваниеСодержимого.СерединаЦентр;

РегуляторВверхВниз1 = Форма1.ЭлементыУправления.Добавить(Ф.РегуляторВверхВниз());
РегуляторВверхВниз1.Положение = Ф.Точка(Кнопка1.Лево + 35, Надпись1.Низ + 5);
РегуляторВверхВниз1.РазмерШрифта = 18;
РегуляторВверхВниз1.Ширина = РегуляторВверхВниз1.Высота * 2;
РегуляторВверхВниз1.Минимум = 1;
РегуляторВверхВниз1.Максимум = 60;

Панель1 = Форма1.ЭлементыУправления.Добавить(Ф.Панель());
Панель1.Границы = Ф.Прямоугольник(Кнопка1.Право + 20, Кнопка1.Верх, 250, (Кнопка1.Высота * 2) + 10);
Панель1.СтильГраницы = Ф.СтильГраницы.Трехмерная;

Надпись2 = Панель1.ЭлементыУправления.Добавить(Ф.Надпись());
Надпись2.Размер = Ф.Размер(Панель1.Ширина, Панель1.Высота);
Надпись2.Текст = "" + КоличЦикловТаймера;
Надпись2.Шрифт = Ф.Шрифт(, 26, Ф.СтильШрифта.Жирный);
Надпись2.ОсновнойЦвет = Ф.Цвет().Красный;
Надпись2.ВыравниваниеТекста = Ф.ВыравниваниеСодержимого.СерединаЦентр;
Надпись2.Центр();

Таймер1 = Ф.Таймер();
Таймер1.Интервал = РегуляторВверхВниз1.Значение * 1000;
Таймер1.ПриСрабатыванииТаймера = Ф.Действие(ЭтотОбъект, "Таймер1_ПриСрабатыванииТаймера");

Ф.ЗапуститьОбработкуСобытий();
